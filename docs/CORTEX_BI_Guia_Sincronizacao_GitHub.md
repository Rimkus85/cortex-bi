```python
def verify_signature(payload, signature):
    """Verifica assinatura do webhook do GitHub"""
    if not signature:

        WEBHOOK_SECRET.encode(),
```

```python
import sys
import time
import subprocess
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

class CodeChangeHandler(FileSystemEventHandler):
    def __init__(self, restart_callback):
        self.restart_callback = restart_callback
        self.last_restart = 0
        
    def on_modified(self, event):

            if time.time() - self.last_restart > 2:
                print(f"üîÑ Arquivo modificado: {event.src_path}")
                self.restart_callback()
                self.last_restart = time.time()

class DevServer:
    def __init__(self):
        self.process = None
        
    def start(self):
        """Inicia o servidor"""
        print("üöÄ Iniciando C√ìRTEX BI...")
        self.process = subprocess.Popen(
            [sys.executable, 'src/main_ai.py'],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE

    def stop(self):
        """Para o servidor"""
        if self.process:
            print("‚è∏Ô∏è  Parando servidor...")
            self.process.terminate()
            self.process.wait()
            
    def restart(self):
        """Reinicia o servidor"""
        self.stop()
        time.sleep(1)
        self.start()

if __name__ == '__main__':
    server = DevServer()
    server.start()
    
    # Configurar watchdog
    handler = CodeChangeHandler(server.restart)
    observer = Observer()
    observer.schedule(handler, 'src/', recursive=True)
    observer.start()
    
    print("üëÄ Monitorando mudan√ßas em src/...")
    print("Pressione Ctrl+C para parar")

    except KeyboardInterrupt:

        server.stop()
    
    observer.join()
```
